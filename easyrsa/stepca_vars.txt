# --- Identity ---
set_var EASYRSA_REQ_COUNTRY    "US"
set_var EASYRSA_REQ_PROVINCE   "Florida"
set_var EASYRSA_REQ_CITY       "Brandon"
set_var EASYRSA_REQ_ORG        "Home Network Infrastructure"
set_var EASYRSA_REQ_OU         "Certificate Issuing Authority"

# --- Crypto Settings ---
# Must match the Root's algorithm for a clean chain
set_var EASYRSA_ALGO           "ec"
set_var EASYRSA_CURVE          "secp384r1"
set_var EASYRSA_DIGEST         "sha384"

# --- Expiration ---
# Intermediate certs should be shorter than the Root (e.g., 2-5 years)
set_var EASYRSA_CERT_EXPIRE    365  # 1 year
set_var EASYRSA_CRL_DAYS       45   # How often you refresh revocation lists

# --- Security Constraints ---
# Prevents a signed client/server cert from acting as a CA itself
set_var EASYRSA_NS_SUPPORT     "no"
set_var EASYRSA_NS_COMMENT     "Internal Issued Certificate"

# --- Performance ---
# Uses a faster random number generator inside the container
set_var EASYRSA_RAND_SN        "yes"
